// DeepEq

function deepEq(a,b) {
    var type = typeof(a);
    if (type === 'number' || 
        type === 'string' ||
        type === 'boolean') {
        return a === b;
    }
    if (type === 'object') {
        for (field in a) {
            if (!deepEq(a[field], b[field])) {
                return false;
            }
        }
        return true;
    }

    console.log("Error: doing comparison of things that aren't number/string/boolean/object");
    process.exit(1);
}
